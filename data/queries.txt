T1 <- SELECT Salary >= 30000 FROM EMPLOYEE
Q1 <- PROJECT Ssn,Salary FROM T1
EXPORT Q1
CLEAR T1
CLEAR Q1

T1 <- PROJECT Ssn, Super_ssn, Bdate FROM EMPLOYEE
CR <- CROSS T1 T1
T2 <- SELECT T11_Super_ssn == T12_Ssn FROM CR
T3 <- SELECT T11_Bdate == T12_Bdate FROM T2
Q2 <- PROJECT T11_Ssn, T11_Super_ssn, T11_Bdate FROM T3
RENAME T11_Ssn TO Ssn FROM Q2
RENAME T11_Super_ssn TO Super_ssn FROM Q2
RENAME T11_Bdate TO Bdate FROM Q2
EXPORT Q2
CLEAR T1
CLEAR T2
CLEAR T3
CLEAR Q2
CLEAR CR

E1 <- PROJECT Ssn, Sex FROM EMPLOYEE
CR1 <- CROSS WORKS_ON E1
T1 <- SELECT Essn == Ssn FROM CR1
F <- SELECT Sex == 1 FROM T1
M <- SELECT Sex == 0 FROM T1
PF <- PROJECT Pno FROM F
PM <- PROJECT Pno FROM M
BOTHGENDER <- CROSS PF PM
RESULT <- SELECT PF_Pno == PM_Pno FROM BOTHGENDER
Q3 <- PROJECT PF_Pno FROM RESULT
RENAME PF_Pno TO Pno FROM Q3
EXPORT Q3
CLEAR E1
CLEAR CR1
CLEAR T1
CLEAR F
CLEAR M
CLEAR PF
CLEAR PM
CLEAR BOTHGENDER
CLEAR RESULT
CLEAR Q3

E1 <- PROJECT Ssn, Super_ssn FROM EMPLOYEE
CR1 <- CROSS WORKS_ON E1
